\section*{约定}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{LIST}{1cm}

  \IT{\kwd{NAME}; \FU{NAME}; \GFU{NAME}; \MC{NAME}; \SO{NAME}; \V{\A name\A};
    \CNS{name}}
  {
  由 Common Lisp 定义的符号、函数、广义函数、宏、特殊操作符、变量、常量。
  }

  \IT{\VAR{them}\qquad\qquad\qquad}
  {
  实际代码的占位符。
  }

  \IT{\LIT{me}\qquad\qquad\qquad\qquad}
  {
  字面文本。
  }

  \IT{\Op{\VAR{foo}\DF{\LIT{bar}}}\qquad\qquad}
  {
  零个或一个 \VAR{foo}，默认为 \LIT{bar}。
  }

  \IT{\OPn{\VAR{foo}}; \Goos{\VAR{foo}}}
  {
  零个或多个 \VAR{foo}s。
  }

  \IT{\RP{\VAR{foo}}; \Goop{\VAR{foo}}}
  {
  一个或多个 \VAR{foo}s。
  }

  \IT{\VAR{foos}\qquad\qquad\qquad}
  {
  复数代表列表参数。
  }

  \IT{\Goo{\VAR{foo}\XOR\VAR{bar}\XOR\VAR{baz}};
    \xorGOO{
      \VAR{foo}\\
      \VAR{bar}\\
      \VAR{baz}%
    }{.}}
  {
  \VAR{foo} 或 \VAR{bar} 或 \VAR{baz}。
  }

  \IT{\orGOO{
      \VAR{foo}\\
      \VAR{bar}\\
      \VAR{baz} 
    }{.}}
  {
  \VAR{foo}、\VAR{bar} 及 \VAR{baz} 的任意子集。
  }

  \IT{\NEV{\VAR{foo}}\qquad\qquad\qquad}
  {
  不对参数 \VAR{foo} 求值。
  }

  \IT{\DES{\VAR{bar}}\qquad\qquad\qquad}
  {
  可能会修改参数 \VAR{bar}。
  }

  \IT{\PROGN{\VAR{foo}}\qquad\qquad\qquad}
  {
  \OPn{\VAR{foo}} 在 \SO{progn} 内求值；参见第 \pageref{:progn} 页。
  }

  \IT{\retval{\VAR{foo}}; \retvalii{\VAR{bar}};
    \retvaln{\VAR{n}}{\VAR{baz}}}
  {
  第一、第二、第 \VAR{n} 返回值。
  }

  \IT{\T; \NIL\qquad\qquad\qquad}
  {
  \kwd*{T}, 或上下文真值；及 \kwd*{NIL} 或 \kwd{()}。
  }
  
\end{LIST}
